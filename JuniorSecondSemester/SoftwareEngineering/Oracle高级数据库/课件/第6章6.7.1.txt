DECLARE
	v_xh char(6);
	v_zxf number(2);
	CURSOR XS_CUR3
	IS
	SELECT XH,ZXF
		FROM XSB
		WHERE ZY= '¼ÆËã»ú';
	BEGIN
		OPEN XS_CUR3;
		FETCH XS_CUR3 INTO v_xh, v_zxf;
		WHILE XS_CUR3%FOUND
		LOOP
			DBMS_OUTPUT.PUT_LINE (v_xh+v_zxf);
			FETCH XS_CUR3 INTO v_xh, v_zxf;
		END LOOP;
		CLOSE XS_CUR3;
	END;
